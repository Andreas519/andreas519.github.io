# ESP32-espnow-sender.py
import network, espnow, time

wlan_sta = network.WLAN(network.STA_IF)
wlan_sta.active(True)

e = espnow.ESPNow()
e.active(True)
peer = b'\xff\xff\xff\xff\xff\xff'
e.add_peer(peer)
while True:
    e.send(peer, "Hallo vom ESP")
    time.sleep(1.1)  # Sekunden
    print(".", end = "")
