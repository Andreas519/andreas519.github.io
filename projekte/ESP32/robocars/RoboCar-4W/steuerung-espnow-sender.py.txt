import network, espnow

wlan_sta = network.WLAN(network.STA_IF)
wlan_sta.active(True)

e = espnow.ESPNow()
e.active(True)

peer = b'\xff\xff\xff\xff\xff\xff'
e.add_peer(peer)

kennung = "RoboCar-max"
print("\n", kennung)
print(" an :", peer)
cnt = 0  # Zähler für gesendete Packete
while True:
    eingabe = input("zusendende Daten eingeben: ")
    cnt+=1;
    nachricht = kennung + "," + str(cnt) + "," +  eingabe
    e.send(peer, bytes(nachricht,""))
    print("Sende: "+ nachricht)
