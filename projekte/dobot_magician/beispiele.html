Grundlegende Beispiele

Hier finden Sie eine Sammlung von Programmierbeispielen zur Steuerung des Dobot Magician mit Python.

Beispiel 1: Initialisierung und Home-Position

#!/usr/bin/env python3
# Initialisierung und Rückkehr zur Home-Position

from pydobotplus import PyDobot

# Verbindung mit Dobot herstellen
dobot = PyDobot()

# Zur Home-Position fahren
dobot.move_to_home()

# Verbindung schließen
dobot.close()

Beispiel 2: Bewegung zu einer bestimmten Position

#!/usr/bin/env python3
# Bewegung zu einer bestimmten XYZ-Position

from pydobotplus import PyDobot

dobot = PyDobot()

# Bewegung zu Position (200, 100, 50)
# x, y, z in mm
dobot.move_to(200, 100, 50)

# Warte bis die Bewegung abgeschlossen ist
import time
time.sleep(2)

dobot.close()

Beispiel 3: Sauggreifer steuern

#!/usr/bin/env python3
# Sauggreifer aktivieren und deaktivieren

from pydobotplus import PyDobot
import time

dobot = PyDobot()

# Zur Pickup-Position fahren
dobot.move_to(150, 150, 0)
time.sleep(1)

# Sauggreifer aktivieren
dobot.suction_cup(True)
print("Sauggreifer aktiviert")
time.sleep(1)

# Objekt anheben
dobot.move_to(150, 150, 100)
time.sleep(1)

# Sauggreifer deaktivieren
dobot.suction_cup(False)
print("Sauggreifer deaktiviert")
time.sleep(1)

dobot.close()

Beispiel 4: Mehrere Positionen nacheinander anfähren

#!/usr/bin/env python3
# Mehrere Positionen in Sequenz anfähren

from pydobotplus import PyDobot
import time

dobot = PyDobot()

# Liste mit Positionen
positions = [
    (150, 150, 50),
    (200, 100, 50),
    (100, 200, 50),
    (150, 150, 50)
]

for x, y, z in positions:
    print(f"Fahre zu ({x}, {y}, {z})")
    dobot.move_to(x, y, z)
    time.sleep(1)

dobot.close()

Beispiel 5: Endeffector-Winkel einstellen

#!/usr/bin/env python3
# Endeffector-Winkel (Rotation) einstellen

from pydobotplus import PyDobot
import time

dobot = PyDobot()

# Bewegung mit verschiedenen Winkeln
# x, y, z, winkel (in Grad)
dobot.move_to(150, 150, 50, 0)
time.sleep(1)

dobot.move_to(150, 150, 50, 45)
time.sleep(1)

dobot.move_to(150, 150, 50, 90)
time.sleep(1)

dobot.close()

Beispiel 6: Geschwindigkeit einstellen

#!/usr/bin/env python3
# Bewegungsgeschwindigkeit einstellen

from pydobotplus import PyDobot
import time

dobot = PyDobot()

# Langsame Bewegung
dobot.speed(50)
dobot.move_to(200, 100, 50)
time.sleep(2)

# Schnelle Bewegung
dobot.speed(100)
dobot.move_to(100, 200, 50)
time.sleep(1)

dobot.close()
