Dobot Studio – einfache Python-Beispiele
Gemeinsame Hilfsdatei Verbindung testen und aktuelle Position anzeigen Dobot auf Home-Position fahren Einen Punkt anfahren Kleines Rechteck abfahren Stift absenken und Linie zeichnen Sauggreifer ein- und ausschalten Einfaches Pick-and-Place mit Sauggreifer Kleine Schleife: drei Positionen nacheinander

Beispielsammlung für den Dobot Magician mit Python über DobotDllType.
Die Programme laufen auf dem PC und senden Befehle per USB/serieller Verbindung an den Roboter.

Hinweise vor dem Start

  • DobotDllType.py muss für die Python-Programme erreichbar sein.
  • Alle Koordinaten sind Beispielwerte. Besonders der Z-Wert muss an euren Aufbau angepasst werden.
  • Bei echten Bewegungen langsam testen und den Arbeitsbereich frei halten.
  • Die Datei dobot_helper.py sollte im selben Ordner liegen wie die Beispielprogramme.

dobot_helper.py

Gemeinsame Hilfsdatei

import time
import DobotDllType as dType

CON_STR = {
    dType.DobotConnect.DobotConnect_NoError: "DobotConnect_NoError",
    dType.DobotConnect.DobotConnect_NotFound: "DobotConnect_NotFound",
    dType.DobotConnect.DobotConnect_Occupied: "DobotConnect_Occupied"
}


def connect_dobot(port=""):
    """
    port="" versucht automatische Verbindung.
    Bei Problemen z. B. "COM3" eintragen.
    """
    api = dType.load()
    state = dType.ConnectDobot(api, port, 115200)[0]

    print("Connect status:", CON_STR[state])

    if state != dType.DobotConnect.DobotConnect_NoError:
        raise RuntimeError("Dobot konnte nicht verbunden werden.")

    dType.SetCmdTimeout(api, 3000)
    dType.SetQueuedCmdClear(api)

    # Langsame, schulgeeignete Bewegungen
    dType.SetPTPJointParams(api, 100, 100, 100, 100,
                            100, 100, 100, 100, isQueued=1)
    dType.SetPTPCommonParams(api, 50, 50, isQueued=1)

    return api


def run_queue(api, last_index):
    dType.SetQueuedCmdStartExec(api)

    while last_index > dType.GetQueuedCmdCurrentIndex(api)[0]:
        dType.dSleep(100)

    dType.SetQueuedCmdStopExec(api)


def disconnect_dobot(api):
    dType.DisconnectDobot(api)
    print("Dobot getrennt.")

01_position_anzeigen.py

Verbindung testen und aktuelle Position anzeigen

import DobotDllType as dType
from dobot_helper import connect_dobot, disconnect_dobot

api = connect_dobot()

pose = dType.GetPose(api)

print("Aktuelle Position:")
print("X =", round(pose[0], 2))
print("Y =", round(pose[1], 2))
print("Z =", round(pose[2], 2))
print("R =", round(pose[3], 2))

disconnect_dobot(api)

02_home.py

Dobot auf Home-Position fahren

import DobotDllType as dType
from dobot_helper import connect_dobot, run_queue, disconnect_dobot

api = connect_dobot()

# Home-Parameter: X, Y, Z, R
dType.SetHOMEParams(api, 200, 0, 50, 0, isQueued=1)

last_index = dType.SetHOMECmd(api, temp=0, isQueued=1)[0]

run_queue(api, last_index)
disconnect_dobot(api)

06_sauggreifer_test.py

Sauggreifer ein- und ausschalten

import DobotDllType as dType
from dobot_helper import connect_dobot, run_queue, disconnect_dobot

api = connect_dobot()

last_index = 0

# Zu einem Punkt fahren
last_index = dType.SetPTPCmd(
    api,
    dType.PTPMode.PTPMOVLXYZMode,
    220, 0, 40, 0,
    isQueued=1
)[0]

# Sauggreifer einschalten
last_index = dType.SetEndEffectorSuctionCup(
    api,
    1,  # enableCtrl
    1,  # suction on
    isQueued=1
)[0]

# Kurz warten
last_index = dType.SetWAITCmd(api, 1000, isQueued=1)[0]

# Sauggreifer ausschalten
last_index = dType.SetEndEffectorSuctionCup(
    api,
    1,
    0,
    isQueued=1
)[0]

run_queue(api, last_index)
disconnect_dobot(api)

07_pick_and_place.py

Einfaches Pick-and-Place mit Sauggreifer

import DobotDllType as dType
from dobot_helper import connect_dobot, run_queue, disconnect_dobot

api = connect_dobot()

z_oben = 60
z_unten = 25

aufnahme = (220, -40)
ablage = (220, 40)

last_index = 0

# Über Aufnahmepunkt
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             aufnahme[0], aufnahme[1], z_oben, 0, isQueued=1)[0]

# Absenken
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             aufnahme[0], aufnahme[1], z_unten, 0, isQueued=1)[0]

# Ansaugen
last_index = dType.SetEndEffectorSuctionCup(api, 1, 1, isQueued=1)[0]
last_index = dType.SetWAITCmd(api, 500, isQueued=1)[0]

# Anheben
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             aufnahme[0], aufnahme[1], z_oben, 0, isQueued=1)[0]

# Über Ablagepunkt
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             ablage[0], ablage[1], z_oben, 0, isQueued=1)[0]

# Absenken
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             ablage[0], ablage[1], z_unten, 0, isQueued=1)[0]

# Loslassen
last_index = dType.SetEndEffectorSuctionCup(api, 1, 0, isQueued=1)[0]
last_index = dType.SetWAITCmd(api, 500, isQueued=1)[0]

# Wieder hoch
last_index = dType.SetPTPCmd(api, dType.PTPMode.PTPMOVLXYZMode,
                             ablage[0], ablage[1], z_oben, 0, isQueued=1)[0]

run_queue(api, last_index)
disconnect_dobot(api)

08_positionen_schleife.py

Kleine Schleife: drei Positionen nacheinander

import DobotDllType as dType
from dobot_helper import connect_dobot, run_queue, disconnect_dobot

api = connect_dobot()

positionen = [
    (200, -60, 50, 0),
    (230,   0, 50, 0),
    (200,  60, 50, 0),
]

last_index = 0

for wiederholung in range(3):
    print("Durchlauf", wiederholung + 1)

    for x, y, z, r in positionen:
        last_index = dType.SetPTPCmd(
            api,
            dType.PTPMode.PTPMOVLXYZMode,
            x, y, z, r,
            isQueued=1
        )[0]

run_queue(api, last_index)
disconnect_dobot(api)
